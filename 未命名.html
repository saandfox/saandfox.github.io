    <!-- ⚙️ 核心邏輯 JavaScript -->

    <script>

        // 1. 📦 模擬五金行商品資料庫 (您可以自由修改條碼、品名與價格)

        const hardwareDatabase = {

            "4710123456789": { name: "不鏽鋼六角螺絲 (10入)", spec: "M8 x 20mm / SUS304", price: 85, stock: "庫存 50 包" },

            "4719876543210": { name: "多功能不鏽鋼美工刀", spec: "SK5 高碳鋼刀片 / 附自動鎖定", price: 120, stock: "庫存 12 把" },

            "88812345": { name: "高導電絕緣膠帶 (黑)", spec: "0.13mm x 19mm x 20Y", price: 35, stock: "庫存 120 捲" },

            "A101": { name: "牧田款式 充電式衝擊起子機", spec: "18V / 雙鋰電 / 附工具箱", price: 3850, stock: "庫存 3 台" }

        };


        let html5QrcodeScanner = null;

        const resultBox = document.getElementById('result-box');

        const scannedCode = document.getElementById('scanned-code');

        const productInfo = document.getElementById('product-info');

        const productNotFound = document.getElementById('product-not-found');

        

        const startBtn = document.getElementById('start-btn');

        const stopBtn = document.getElementById('stop-btn');


        // 2. 📸 當掃描成功時的處理邏輯

        function onScanSuccess(decodedText, decodedResult) {

            // 播放一聲簡短的提示音（可選）

            playBeep();

            

            // 顯示結果區塊與條碼

            resultBox.classList.remove('hidden');

            scannedCode.innerText = decodedText;


            // 比對資料庫

            const product = hardwareDatabase[decodedText];


            if (product) {

                // 找到商品，填入欄位

                document.getElementById('prod-name').innerText = product.name;

                document.getElementById('prod-spec').innerText = `規格：${product.spec}`;

                document.getElementById('prod-price').innerText = `NT$ ${product.price}`;

                document.getElementById('prod-stock').innerText = product.stock;

                

                productInfo.classList.remove('hidden');

                productNotFound.classList.add('hidden');

            } else {

                // 找不到商品

                productInfo.classList.add('hidden');

                productNotFound.classList.remove('hidden');

            }

        }


        // 3. 🛠️ 初始化與控制相機

        function startScanner() {

            html5QrcodeScanner = new Html5Qrcode("reader");

            

            // 設定掃描參數：優先使用後置相機 (facingMode: "environment")

            const config = { 

                fps: 15, 

                qrbox: (width, height) => {

                    return { width: width * 0.8, height: height * 0.4 } // 寬扁型框，極度適合五金常見的一維長條碼

                },

                aspectRatio: 1.0

            };


            html5QrcodeScanner.start(

                { facingMode: "environment" }, 

                config, 

                onScanSuccess

            ).then(() => {

                startBtn.classList.add('hidden');

                stopBtn.classList.remove('hidden');

            }).catch((err) => {

                alert("無法開啟相機，請檢查瀏覽器是否已授權相機權限！");

                console.error(err);

            });

        }


        function stopScanner() {

            if (html5QrcodeScanner) {

                html5QrcodeScanner.stop().then(() => {

                    html5QrcodeScanner.clear();

                    startBtn.classList.remove('hidden');

                    stopBtn.classList.add('hidden');

                }).catch((err) => {

                    console.error("關閉相機失敗", err);

                });

            }

        }


        // 監聽按鈕點擊

        startBtn.addEventListener('click', startScanner);

        stopBtn.addEventListener('click', stopScanner);


        // 🎵 輔助功能：掃描成功的嗶聲效果

        function playBeep() {

            const context = new (window.AudioContext || window.webkitAudioContext)();

            const oscillator = context.createOscillator();

            oscillator.type = 'sine';

            oscillator.frequency.setValueAtTime(1000, context.currentTime); // 1000Hz 高音

            oscillator.connect(context.destination);

            oscillator.start();

            oscillator.stop(context.currentTime + 0.1); // 響 0.1 秒

        }
